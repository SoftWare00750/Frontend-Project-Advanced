let & const // ES6 variable declarations ES6
let-const.js 
// const — value cannot be reassigned
const STORE_NAME = "Student Tech Store";
const TAX_RATE   = 0.075;

// let — value can be reassigned
let cartTotal  = 0;
let activeUser = "Student";

// reassign let — OK
cartTotal  = 250;
activeUser = "Ada";

// reassign const — will throw TypeError
// STORE_NAME = "Other";  ← ❌ Error!

console.log(STORE_NAME, TAX_RATE, cartTotal, activeUser);
// Click ▶ Run to execute
const is for values that should never change (constants). let is for values that update over time (counters, state). Never use var in modern JS.
Scope // block vs function scope Scope

BLOCK SCOPE (let / const)

let price = 100;    // outer

{
  let price = 80;  // inner
  console.log(price);  // 80
}

console.log(price);    // 100

FUNCTION SCOPE (var — avoid)

var discount = 10;   // outer

{
  var discount = 25; // overwrites!
}

console.log(discount); // 25 ⚠️

let and const are block-scoped — they stay inside {}. var leaks out of blocks (only respects functions), which causes bugs. Always prefer let and const.

Array — reduce() // accumulate to a single value Array
reduce() collapses an array into a single value by running an accumulator function on each element.
reduce.js 
// Sum all prices
const totalValue = products.reduce(
  (acc, p) => acc + p.price, 0
);

// Total stock count
const totalStock = products.reduce(
  (acc, p) => acc + p.stock, 0
);

// Group by category
const byCategory = products.reduce((acc, p) => {
  acc[p.category] = (acc[p.category] || 0) + 1;
  return acc;
}, {});

console.log("Total value:", `₦${totalValue.toLocaleString()}`);
console.log("Total stock:", totalStock);
console.log("By category:", byCategory);

Array — forEach() // side-effects on each element Array
forEach.js 
// Log each item in the cart
const cart = [
  { name: "MacBook Air", qty: 1, price: 450000 },
  { name: "USB Hub",     qty: 2, price: 8000   },
  { name: "Mouse",       qty: 1, price: 12000  }
];

let receiptTotal = 0;

cart.forEach((item, index) => {
  const lineTotal = item.price * item.qty;
  receiptTotal += lineTotal;
  console.log(`${index+1}. ${item.name} x${item.qty} = ₦${lineTotal.toLocaleString()}`);
});

console.log("TOTAL: ₦" + receiptTotal.toLocaleString());
// Click ▶ Run to execute
forEach() is for side effects (logging, updating DOM, saving data). It always returns undefined — never use it when you need a new array. Use map() or filter() for that.
Objects // properties, methods, destructuring Objects
objects.js 
// Object literal
const product = {
  id: 1,
  name: "MacBook Air M2",
  price: 450000,
  category: "laptop",
  inStock: true,
  // Method shorthand (ES6)
  priceWithTax() { return this.price * 1.075; },
  summary()      { return `${this.name} — ₦${this.price.toLocaleString()}`; }
};

// Destructuring
const { name, price, category } = product;

// Spread operator — clone & update
const discounted = { ...product, price: 400000 };

console.log(product.summary());
console.log("Tax price:", product.priceWithTax());
console.log("Destructured:", name, price, category);
console.log("Discounted:", discounted.price);

  `${p.name.padEnd(22)} ${formatPrice(p.price)}`
));

// ── FILTER — budget under ₦100,000 ────────────────────

  .filter(p => isAffordable(p, 100000))
  .map(p => p.name);

const storeValue   = products.reduce((s, p) => s + p.price, 0);
const avgPrice     = storeValue / products.length;
